function handleAIWriteIntent(aiText) {
  // 1) Parse patches
  var patches = PatchParser.parse(aiText);

  // fallback: if no patches, keep existing file-write logic
  if (!patches || !patches.length) {
    return detectAndOfferFileWrite(aiText); // your existing path
  }

  // 2) Validate patches
  var valid = [];
  var rejected = [];

  for (var i = 0; i < patches.length; i++) {
    if (PatchValidator.validate(patches[i])) valid.push(patches[i]);
    else rejected.push(patches[i]);
  }

  if (!valid.length) return; // nothing to do

  // 3) Score + route
  routePatchesByConfidence(valid);
}
3) Confidence Routing Layer
function routePatchesByConfidence(patches) {
  var results = [];

  for (var i = 0; i < patches.length; i++) {
    var p = patches[i];

    var score = ConfidenceEngine.scorePatch(p, {
      fileCount: patches.length,
      recentErrors: sessionMemory.recentErrors || []
    });

    results.push({
      patch: p,
      score: score,
      level: ConfidenceEngine.classify(score)
    });
  }

  // Split buckets
  var safe = [];
  var review = [];
  var risky = [];

  for (var j = 0; j < results.length; j++) {
    var r = results[j];
    if (r.level === "safe") safe.push(r);
    else if (r.level === "review") review.push(r);
    else risky.push(r);
  }

  // --- AUTO EXECUTE SAFE ---
  if (safe.length) {
    try {
      var safeFiles = convertPatchesToFiles(safe);
      executeTransactionWithLogging(safeFiles, "auto");

      appendTimelineItem({
        type: "auto-commit",
        count: safeFiles.length
      });

    } catch (err) {
      var norm = normalizeError(err);
      PersistentMemory.logError(norm);
    }
  }

  // --- SEND OTHERS TO UI ---
  var pending = review.concat(risky);
  if (pending.length) {
    var files = convertPatchesToFiles(pending);

    // attach scores for UI
    for (var k = 0; k < pending.length; k++) {
      files[k]._confidence = pending[k].score;
      files[k]._level = pending[k].level;
    }

    offerGroupedFileWrites(files); // your existing UI
  }
}
4) Patch → File Converter
function convertPatchesToFiles(results) {
  var files = [];

  for (var i = 0; i < results.length; i++) {
    var p = results[i].patch;

    var resolved = resolveProjectWritePath(p.file);
    var original = fsApi.readFile(resolved);

    var updated = PatchApplier.apply(original, p);

    files.push({
      path: p.file,
      content: updated
    });
  }

  return files;
}
5) Transaction Wrapper (with memory + audit)
function executeTransactionWithLogging(files, mode) {
  try {
    var resolvedFiles = files.map(function(f) {
      return {
        path: resolveProjectWritePath(f.path),
        content: f.content
      };
    });

    TransactionManager.run({ files: resolvedFiles });

    PersistentMemory.logTransaction({
      files: resolvedFiles.map(function(f){ return f.path; }),
      status: "committed",
      mode: mode || "manual"
    });

  } catch (err) {
    var normalized = normalizeError(err);

    PersistentMemory.logTransaction({
      files: files.map(function(f){ return f.path; }),
      status: "failed",
      error: normalized.message
    });

    throw err;
  }
}

What it does

When a patch fails:

AI patch → validator/transaction fails → capture error → build retry prompt → AI returns corrected patch → re-score → execute again

So instead of stopping at failure, the engine gets one controlled recovery pass.

Why it is a game changer

It converts the system from:

suggest → fail → wait for user

to:

suggest → fail → self-correct → retry safely

That is the first real step into governed autonomy.

Strict rule set

Keep it tightly controlled:

max 1 auto-retry at first
retry only for safe failure types
retry must still pass:
patch validation
confidence scoring
transaction safeguards
no infinite loops
no retry on dangerous categories
Retryable failures

Allow retry for things like:

syntax error
malformed patch block
target function not found
small patch apply mismatch
validator formatting issue

Do not auto-retry for:

path traversal
blocked APIs
policy/security violations
repeated risky patches
rollback corruption states
Core flow
1. AI returns patch
2. Patch parsed + scored
3. Execute transaction
4. If failed with retryable error:
   - log failure
   - build compact retry context
   - ask AI for corrected patch only
   - parse + validate + re-score
   - if safe/review: execute
5. If retry fails:
   - send to grouped UI/manual review
