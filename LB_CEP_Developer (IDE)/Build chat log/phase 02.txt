function parsePatchBlocks(text) {
  var blocks = [];
  var regex = /\/\/ PATCH: (.*?)\n([\s\S]*?)\/\/ ---\n([\s\S]*?)\/\/ ---/g;
  var match;

  while ((match = regex.exec(text)) !== null) {

      meta: match[2],
      content: match[3]
    });
  }

}

// js/modules/memory/persistentMemory.js
(function (global) {
  "use strict";

  var MAX_ITEMS = 50;        // hard cap
  var MAX_SUMMARY = 10;      // items used for AI context
  var MEMORY_FILE = ".letterblack_memory.ndjson";

  var state = {
    entries: []
  };

  // --- Utils ---
  function now() {
    return Date.now();
  }

  function safeResolve(path) {
    try {
      return resolveProjectWritePath(path); // existing sandbox
    } catch (e) {
      return null;
    }
  }

  function trim(arr, max) {
    while (arr.length > max) arr.shift();
  }

  function parseNDJSON(text) {
    if (!text) return [];
    var lines = text.split("\n");
    var out = [];
    for (var i = 0; i < lines.length; i++) {
      var line = lines[i].trim();
      if (!line) continue;
      try {
        out.push(JSON.parse(line));
      } catch (e) {
        // ignore malformed lines
      }
    }
    return out;
  }

  function toNDJSON(entries) {
    var out = [];
    for (var i = 0; i < entries.length; i++) {
      out.push(JSON.stringify(entries[i]));
    }
    return out.join("\n") + "\n";
  }

  // --- IO ---
  function readMemoryFile() {
    var resolved = safeResolve(MEMORY_FILE);
    if (!resolved) return [];
    try {
      if (!fsApi.exists(resolved)) return [];
      var text = fsApi.readFile(resolved);
      return parseNDJSON(text);
    } catch (e) {
      return [];
    }
  }

  function writeMemoryFile(entries) {
    var resolved = safeResolve(MEMORY_FILE);
    if (!resolved) return;
    try {
      fsApi.writeFile(resolved, toNDJSON(entries));
    } catch (e) {
      // swallow; memory should never break core flow
    }
  }

  // --- Public: load ---
  function load() {
    state.entries = readMemoryFile();
    trim(state.entries, MAX_ITEMS);
  }

  // --- Public: log ---
  function log(entry) {
    try {
      entry.time = now();
      state.entries.push(entry);
      trim(state.entries, MAX_ITEMS);
      writeMemoryFile(state.entries);
    } catch (e) {
      // non-blocking
    }
  }

  // --- Public: helpers ---
  function logError(err) {
    log({
      type: "error",
      message: err && err.message ? err.message : String(err)
    });
  }

  function logTransaction(info) {
    log({
      type: "transaction",
      files: info.files || [],
      status: info.status || "unknown",
      error: info.error || null
    });
  }

  // --- Summary builder for AI ---
  function buildSummary() {
    var last = state.entries.slice(-MAX_SUMMARY);

    var errors = {};
    var files = {};
    var notes = [];

    for (var i = 0; i < last.length; i++) {
      var e = last[i];

      if (e.type === "error" && e.message) {
        errors[e.message] = (errors[e.message] || 0) + 1;
      }

      if (e.type === "transaction" && e.files) {
        for (var j = 0; j < e.files.length; j++) {
          var f = e.files[j];
          files[f] = (files[f] || 0) + 1;
        }

        if (e.status === "failed" && e.error) {
          notes.push("Failure: " + e.error);
        }
      }
    }

    // build compact text block
    var lines = [];

    var errKeys = Object.keys(errors);
    if (errKeys.length) {
      lines.push("Recent Errors:");
      for (var k = 0; k < errKeys.length; k++) {
        lines.push("- " + errKeys[k] + " (" + errors[errKeys[k]] + ")");
      }
    }

    var fileKeys = Object.keys(files);
    if (fileKeys.length) {
      lines.push("Recent Files:");
      for (var m = 0; m < fileKeys.length; m++) {
        lines.push("- " + fileKeys[m]);
      }
    }

    for (var n = 0; n < notes.length && n < 5; n++) {
      lines.push(notes[n]);
    }

    return lines.join("\n");
  }

  // --- Expose ---
  global.PersistentMemory = {
    load: load,
    log: log,
    logError: logError,
    logTransaction: logTransaction,
    buildSummary: buildSummary
  };

})(this);

(function (global) {
  "use strict";

  function scorePatch(patch, context) {
    var score = 100;

    // Risk deductions
    if (patch.content.length > 2000) score -= 30;
    if (patch.content.includes("eval(")) score -= 50;
    if (patch.content.includes("innerHTML")) score -= 20;
    if (patch.content.includes("import ")) score -= 40;

    // Multi-file penalty
    if (context.fileCount > 3) score -= 20;

    // History-based penalty
    if (context.recentErrors && context.recentErrors.includes("SYNTAX_ERROR")) {
      score -= 10;
    }

    // Reward small targeted patches
    if (patch.type === "function") score += 10;

    return Math.max(0, Math.min(100, score));
  }

  function classify(score) {
    if (score >= 80) return "safe";
    if (score >= 50) return "review";
    return "risky";
  }

  global.ConfidenceEngine = {
    scorePatch: scorePatch,
    classify: classify
  };

})(this);

function convertPatchesToFiles(results) {
  var files = [];

    var p = results[i].patch;

    var resolved = resolveProjectWritePath(p.file);

function executeTransactionWithLogging(files, mode) {

}
