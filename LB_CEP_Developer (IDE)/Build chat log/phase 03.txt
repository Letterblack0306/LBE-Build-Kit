6.3 Stage backups
function stageAllWrites(txn, fsApi) {
  txn.status = "staging";

  txn.files.forEach(file => {
    if (fsApi.exists(file.absolutePath)) {
      file.originalExists = true;
      file.originalContent = fsApi.readFile(file.absolutePath, "utf8");
      file.originalHash = hashString(file.originalContent);

      const backupPath = getBackupPath(txn, file.relativePath);
      fsApi.ensureDir(dirname(backupPath));
      fsApi.writeFile(backupPath, file.originalContent, "utf8");
    } else {
      file.originalExists = false;
    }

    file.status = "staged";
  });

  persistTransactionManifest(txn, fsApi);
  return txn;
}
6.4 Commit writes
function commitAllWrites(txn, fsApi) {
  const committed = [];

  try {
    for (const file of txn.files) {
      fsApi.ensureDir(dirname(file.absolutePath));
      fsApi.writeFile(file.absolutePath, file.proposedContent, "utf8");
      file.status = "committed";
      committed.push(file.relativePath);
    }

    txn.status = "committed";
    return txn;
  } catch (err) {
    txn.error = normalizeError(err);
    rollbackCommittedWrites(txn, fsApi, committed);
    throw err;
  }
}
6.5 Rollback
function rollbackCommittedWrites(txn, fsApi, committedPaths) {
  txn.rollback.performed = true;

  for (const relativePath of committedPaths.slice().reverse()) {
    const file = txn.files.find(f => f.relativePath === relativePath);

    try {
      if (file.originalExists) {
        const backupPath = getBackupPath(txn, relativePath);
        const backupContent = fsApi.readFile(backupPath, "utf8");
        fsApi.writeFile(file.absolutePath, backupContent, "utf8");
      } else {
        if (fsApi.exists(file.absolutePath)) {
          fsApi.unlink(file.absolutePath);
        }
      }

      file.status = "rolled_back";
      txn.rollback.restoredFiles.push(relativePath);
    } catch (restoreErr) {
      txn.rollback.failedRestores.push({
        relativePath,
        error: normalizeError(restoreErr)
      });
    }
  }

  txn.status = txn.rollback.failedRestores.length ? "failed" : "rolled_back";
}
7) Recommended Execution Wrapper

Create one orchestrator instead of spreading logic through UI handlers.

executeWriteTransaction()
function executeWriteTransaction({ projectRoot, actor, source, files, fsApi, audit, timeline }) {
  const txn = beginWriteTransaction({ projectRoot, actor, files, source });

  try {
    preflightValidateAll(txn, { projectRoot });
    audit.logIntent(txn);
    timeline.pushTransactionStart(txn);

    stageAllWrites(txn, fsApi);
    audit.logStage(txn);

    commitAllWrites(txn, fsApi);
    audit.logCommit(txn);
    timeline.pushTransactionSuccess(txn);

    finalizeTransactionArtifacts(txn, fsApi);
    return txn;
  } catch (err) {
    const normalized = normalizeError(err);
    txn.error = normalized;

    audit.logFailure(txn);
    if (txn.rollback.performed) {
      audit.logRollback(txn);
      timeline.pushTransactionRollback(txn);
    } else {
      timeline.pushTransactionFailure(txn);
    }

    finalizeTransactionArtifacts(txn, fsApi);
    throw normalized;
  }
}

function auditEventToTimelineItem(event) {
  switch (event.eventType) {
    case "txn.intent":
      return {
        type: "write_intent",
        time: event.timestamp,
        label: `Proposed ${event.summary.fileCount} file write${event.summary.fileCount > 1 ? "s" : ""}`,
        txnId: event.txnId,
        status: "pending"
      };

    case "txn.commit_completed":
      return {
        type: "write_commit",
        time: event.timestamp,
        label: `Committed ${event.summary.fileCount} file write${event.summary.fileCount > 1 ? "s" : ""}`,
        txnId: event.txnId,
        status: "success"
      };

    case "txn.rollback_completed":
      return {
        type: "write_rollback",
        time: event.timestamp,
        label: `Rolled back transaction`,
        txnId: event.txnId,
        status: "warning"
      };

    case "txn.failed":
      return {
        type: "write_failed",
        time: event.timestamp,
        label: event.payload && event.payload.error
          ? event.payload.error.message
          : "Transaction failed",
        txnId: event.txnId,
        status: "error"
      };

    default:
      return null;
  }
}

  if (!projectRoot) {
    throw createStructuredError("system", "NO_PROJECT", "No project open");
  }

  txn.files.forEach(file => {
    const resolved = resolveProjectWritePath(projectRoot, file.relativePath);
    file.absolutePath = resolved.absolutePath;
  });

  txn.validation = {
    sandbox: true,
    projectOpen: true,
    conflicts: detectConflicts(txn.files)
  };

  if (txn.validation.conflicts.length) {
    throw createStructuredError("system", "CONFLICT", "Duplicate file paths");
  }

  txn.status = TXN_STATUS.VALIDATED;
  return txn;
}

function detectConflicts(files) {
  const seen = new Set();
  const conflicts = [];

  files.forEach(f => {
    if (seen.has(f.relativePath)) {
      conflicts.push(f.relativePath);
    }
    seen.add(f.relativePath);
  });

  return conflicts;
}

export function stageAllWrites(txn, fsApi) {
  txn.status = TXN_STATUS.STAGING;

  txn.files.forEach(file => {
    if (fsApi.exists(file.absolutePath)) {
      file.originalExists = true;
      file.originalContent = fsApi.readFile(file.absolutePath, "utf8");

      const backupPath = getBackupPath(txn.projectRoot, txn.id, file.relativePath);

      fsApi.ensureDir(dirname(backupPath));
      fsApi.writeFile(backupPath, file.originalContent, "utf8");
    } else {
      file.originalExists = false;
    }

    file.status = FILE_STATUS.STAGED;
  });

  return txn;
}

8) transactionRollback.js
import { FILE_STATUS, TXN_STATUS } from "./transactionTypes.js";
import { getBackupPath } from "./transactionPaths.js";

export function rollbackCommittedWrites(txn, fsApi, committedPaths) {
  txn.rollback.performed = true;

  for (const path of committedPaths.slice().reverse()) {
    const file = txn.files.find(f => f.relativePath === path);

    try {
      if (file.originalExists) {
        const backupPath = getBackupPath(txn.projectRoot, txn.id, path);
        const content = fsApi.readFile(backupPath, "utf8");
        fsApi.writeFile(file.absolutePath, content, "utf8");
      } else {
        if (fsApi.exists(file.absolutePath)) {
          fsApi.unlink(file.absolutePath);
        }
      }

      file.status = FILE_STATUS.ROLLED_BACK;
    } catch (e) {
      txn.rollback.failedRestores.push(path);
    }
  }

  txn.status = txn.rollback.failedRestores.length
    ? TXN_STATUS.FAILED
    : TXN_STATUS.ROLLED_BACK;
}

11) transactionManager.js (CORE ENTRY POINT)
import { createTxnId } from "./transactionTypes.js";
import { preflightValidateAll } from "./transactionValidator.js";
import { stageAllWrites } from "./transactionStaging.js";
import { commitAllWrites } from "./transactionCommit.js";
import { rollbackCommittedWrites } from "./transactionRollback.js";
import { logIntent, logCommit, logFailure, logRollback } from "./transactionAudit.js";

export function executeWriteTransaction(ctx) {
  const txn = createTransaction(ctx);

  try {
    preflightValidateAll(txn, ctx);
    logIntent(txn, ctx);

    stageAllWrites(txn, ctx.fsApi);

    const { committed } = commitAllWrites(txn, ctx.fsApi);

    logCommit(txn, ctx);
    return txn;

  } catch (err) {
    const normalized = ctx.normalizeError(err);
    txn.error = normalized;

    if (err.committed && err.committed.length) {
      rollbackCommittedWrites(txn, ctx.fsApi, err.committed);
      logRollback(txn, ctx);
    }

    logFailure(txn, ctx);
    throw normalized;
  }
}

function createTransaction(ctx) {
  return {
    id: createTxnId(),
    projectRoot: ctx.projectRoot,
    createdAt: Date.now(),
    status: "pending",
    files: ctx.files.map(f => ({
      relativePath: f.relativePath,
      proposedContent: f.content,
      status: "pending"
    })),
    rollback: {
      performed: false,
      failedRestores: []
    },
    error: null
  };
}
